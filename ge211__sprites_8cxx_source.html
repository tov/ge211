149 Circle_sprite::Circle_sprite(int radius, Color color)
150  : Render_sprite{compute_circle_dimensions(radius)}
151 {
152  const int cx = radius;
153  const int cy = radius;
154 
155  for (int y = 0; y < radius; ++y) {
156  for (int x = 0; x < radius; ++x) {
157  if (x * x + y * y < radius * radius) {
158  set_pixel({cx + x, cy + y}, color);
159  set_pixel({cx + x, cy - y - 1}, color);
160  set_pixel({cx - x - 1, cy + y}, color);
161  set_pixel({cx - x - 1, cy - y - 1}, color);
162  }
163  }
164  }

195 Texture
196 Text_sprite::create_texture(const Builder& config)
197 {
198  SDL_Surface* raw;
199 
200  std::string message = config.message();
201 
202  if (message.empty())
203  return Texture{};
204 
205  if (config.word_wrap() > 0) {
206  raw = TTF_RenderUTF8_Blended_Wrapped(
207  config.font().get_raw_(),
208  message.c_str(),
209  config.color().to_sdl_(),
210  static_cast<uint32_t>(config.word_wrap()));
211  } else {
212  auto render = config.antialias() ?
213  &TTF_RenderUTF8_Blended :
214  &TTF_RenderUTF8_Solid;
215  raw = render(config.font().get_raw_(),
216  message.c_str(),
217  config.color().to_sdl_());
218  }
219 
220  if (!raw)
221  throw Host_error{"Could not render text: “" + message + "”"};
222  else
223  return Texture{raw};
224 }
