41 void Engine::run()
42 {
43  SDL_Event e;
44  Sprite_set sprites;
45 
46  bool has_vsync = renderer_.is_vsync();
47 
48  try {
49  game_.on_start();
50 
51  while (!game_.quit_) {
52  handle_events_(e);
53  game_.on_frame(game_.get_prev_frame_length().seconds());
54  game_.poll_channels_();
55  game_.draw(sprites);
56 
57  renderer_.set_color(game_.background_color);
58  renderer_.clear();
59  paint_sprites_(sprites);
60 
61  game_.mark_present_();
62  renderer_.present();
63 
64  Duration allowed_frame_length =
65  (is_focused_ && has_vsync)?
66  min_frame_length : software_frame_length;
67 
68  auto frame_length = game_.frame_start_.elapsed_time();
69  if (frame_length < allowed_frame_length) {
70  auto duration = allowed_frame_length - frame_length;
71  duration.sleep_for();
72  game_.mark_frame_();
73  internal::logging::debug()
74  << "Software vsync slept for "
75  << duration.seconds() << " s";
76  } else {
77  game_.mark_frame_();
78  }
79  }
80 
81  game_.on_quit();
82  } catch (const Exception_base& e) {
83  internal::logging::fatal()
84  << "Uncaught exception:\n "
85  << e.what();
86  exit(1);
87  }

90 void Engine::handle_events_(SDL_Event& e)
91 {
92  while (SDL_PollEvent(&e) != 0) {
93  switch (e.type) {
94  case SDL_QUIT:
95  game_.quit();
96  break;
97 
98  case SDL_TEXTINPUT: {
99  const char* str = e.text.text;
100  const char* end = str + std::strlen(str);
101 
102  while (str < end) {
103  uint32_t code = utf8::next(str, end);
104  if (code) game_.on_key(Key{code});
105  }
106 
107  break;
108  }
109 
110  case SDL_KEYDOWN: {
111  Key key(e.key);
112  if (!e.key.repeat) {
113  game_.on_key_down(key);
114  }
115  if (!key.is_textual()) {
116  game_.on_key(key);
117  }
118  break;
119  }
120 
121  case SDL_KEYUP:
122  game_.on_key_up(Key{e.key});
123  break;
124 
125  case SDL_MOUSEBUTTONDOWN: {
126  Mouse_button button;
127  if (map_button(e.button.button, button))
128  game_.on_mouse_down(button, {e.button.x, e.button.y});
129  break;
130  }
131 
132  case SDL_MOUSEBUTTONUP: {
133  Mouse_button button;
134  if (map_button(e.button.button, button))
135  game_.on_mouse_up(button, {e.button.x, e.button.y});
136  break;
137  }
138 
139  case SDL_MOUSEMOTION:
140  game_.on_mouse_move({e.motion.x, e.motion.y});
141  break;
142 
143  case SDL_WINDOWEVENT:
144  switch (e.window.event) {
145  case SDL_WINDOWEVENT_FOCUS_GAINED:
146  is_focused_ = true;
147  break;
148 
149  case SDL_WINDOWEVENT_FOCUS_LOST:
150  is_focused_ = false;
151  break;
152 
153  default:
154  ;
155  }
156  break;
157 
158  default:
159  ;
160  }
161  }
162 }
